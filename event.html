<script>
    var calendar = new CustomCalendar('#calendar', {
        event: {
            //이벤트 종류별 설정이 필요합니다. items 배열 안에 객체 형태로 이벤트 카테고리에대해서 설정할 수 있습니다.
            items: [{
                title : '카테고리',
                key : 'category',
                type: 'category', //`default`, `categroy`, `select`형식이 있습니다. 가장 기본이 되는 카테고리의 경우 캘린더 핀에 등록됩니다. 
                prompt: '카테고리를 선택해주세요.',
                options : ['카테고리1', '카테고리2'] //`category`, `select` 타입의 경우 선택가능한 options의 설정이 필요합니다.
            },
            {
                title : '운동종목',
                key : 'health',
                type: 'select',
                prompt: '운동 종목을 선택해주세요.',
                options : ['머리','등','허리','다리']
            }],
        },
        options: {
            useDateEvent: true, //이벤트 사용여부
            useDragRange: true, //드래그 사용여부
            eventAuth: true //이벤트 등록 권한
        }
    });

    //이벤트 데이터 설정 사용 예시
    calendar.setEventData = function(){
        //받아올 json 데이터 예시
        var value = [{
            content: {
                title: '이벤트1',
                category: '카테고리1',
                health: '등',
            },
            startDate : '2023-05-31',
            endDate: '2023-06-09',
            state: 'able'
        },{
            content: {
                title: '이벤트2',
                category: '카테고리2',
                health: '허리',
            },
            startDate : '2023-06-07',
            endDate: '2023-06-13',
            state: 'able'
        }];
        return value;
    }
    calendar.initEventData(); //실행
    //calendar.getEventData() //등록된 이벤트 값 가져오기
</script>

이 사이트는 Screen Size - 600px까지 지원하고 있습니다.
